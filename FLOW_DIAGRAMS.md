## 5ï¸âƒ£ Role-Based Access (RLS)

```sql
-- Customer can only see own bookings
CREATE POLICY "customers_own_bookings" ON bookings
  FOR SELECT USING (customer_id = auth.uid());

-- Vendor can see bookings for their salons
CREATE POLICY "vendors_salon_bookings" ON bookings
  FOR SELECT USING (
    EXISTS (
      SELECT 1 FROM salons
      WHERE id = bookings.salon_id
        AND vendor_id = auth.uid()
    )
  );

-- RM can see salons they added
CREATE POLICY "rm_own_salons" ON salons
  FOR SELECT USING (rm_id = auth.uid());

-- Admin can see everything
CREATE POLICY "admin_all_access" ON bookings
  FOR ALL USING (
    EXISTS (
      SELECT 1 FROM profiles
      WHERE id = auth.uid()
        AND role = 'admin'
    )
  );

-- Public can see active salons only
CREATE POLICY "public_active_salons" ON salons
  FOR SELECT USING (
    is_active = true
    AND registration_fee_paid = true
  );
```
